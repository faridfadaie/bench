#!/bin/bash
  for i in $( ls /log_files/*.log); do
    echo processing $i
    X=`echo $i | sed 's/^.*bench.prod-i-[^-]*-\([^-]*\)-[0-9]*\.log$/\1/'`
    D=`echo $i | sed 's/^.*bench.prod-i-[^-]*-[^-]*-\([0-9]*\)\.log$/\1/' | cut -c 1-8`
    H=`echo $i | sed 's/^.*bench.prod-i-[^-]*-[^-]*-\([0-9]*\)\.log$/\1/' | cut -c 9-10`
    /usr/local/bin/s3cmd put -r $i s3://bleepbench/$X/$D/$H/
    rm $i
  done
